        // Enhanced text-to-speech function

        function speakMessage(text, mood) {

            if (!('speechSynthesis' in window)) return;

            

            // Cancel any ongoing speech

            speechSynthesis.cancel();

            

            // Clean text for more natural speech (remove emojis and clean up)

            let cleanText = text.replace(/[🎉🚀💪🌟🔍✨🔥🤔💻⚡🎯👋]/g, '');

            cleanText = cleanText.replace(/!/g, '.'); // Replace ! with . for more natural pauses

            

            const utterance = new SpeechSynthesisUtterance(cleanText);

            

            // Voice settings based on mood

            const voiceSettings = {

                encouraging: { rate: 0.85, pitch: 1.2, volume: 0.8 },

                celebrating: { rate: 1.1, pitch: 1.4, volume: 0.9 },

                nudging: { rate: 0.75, pitch: 1.0, volume: 0.7 },

                worried: { rate: 0.7, pitch: 0.9, volume: 0.6 },

                excited: { rate: 1.2, pitch: 1.5, volume: 1.0 }

            };

            

            const settings = voiceSettings[mood] || voiceSettings.encouraging;

            utterance.rate = settings.rate;

            utterance.pitch = settings.pitch;

            utterance.volume = settings.volume;

            

            // Try to find a more natural voice

            const voices = speechSynthesis.getVoices();

            

            // Prefer these voice types in order

            const preferredVoices = [

                'Google US English Female',

                'Microsoft Zira Desktop',

                'Alex',

                'Samantha',

                'Victoria',

                'Google UK English Female',

                'Google UK English Male',

                'Microsoft David Desktop',

                'Google US English Male'

            ];

            

            let selectedVoice = null;

            for (let preferred of preferredVoices) {

                selectedVoice = voices.find(voice => voice.name.includes(preferred));

                if (selectedVoice) break;

            }

            

            // Fallback to any English voice that sounds natural

            if (!selectedVoice) {

                selectedVoice = voices.find(voice => 

                    voice.lang.startsWith('en') && 

                    (voice.name.includes('Female') || voice.name.includes('Male'))

                );

            }

            

            // Final fallback to default English voice

            if (!selectedVoice) {

                selectedVoice = voices.find(voice => voice.lang.startsWith('en'));

            }

            

            if (selectedVoice) {

                utterance.voice = selectedVoice;

            }

            

            // Add natural pauses and emphasis

            utterance.onboundary = function(event) {

                if (event.name === 'sentence') {

                    // Add slight pause between sentences

                    setTimeout(() => {}, 200);

                }

            };

            

            speechSynthesis.speak(utterance);

        }

        // GitHub API integration

        async function fetchGitHubCommits(repo) {

            try {

                const [owner, repoName] = repo.split('/');

                const response = await fetch(`https://api.github.com/repos/${owner}/${repoName}/commits?per_page=10`);

                

                if (!response.ok) {

                    throw new Error(`HTTP ${response.status}: ${response.statusText}`);

                }

                

                const commits = await response.json();

                return commits;

            } catch (error) {

                console.error('GitHub API Error:', error);

                throw error;

            }

        }


        // Analyze commits and generate personality response

        function analyzeCommits(commits) {

            if (!commits || commits.length === 0) {

                return {

                    message: "Hmm, I don't see any commits yet. Ready to start coding? 🚀",

                    mood: "nudging"

                };

            }


            const now = new Date();

            const recent = commits.filter(commit => {

                const commitDate = new Date(commit.commit.author.date);

                const hoursAgo = (now - commitDate) / (1000 * 60 * 60);

                return hoursAgo < 24; // Last 24 hours

            });


            const lastCommit = commits[0];

            const lastCommitDate = new Date(lastCommit.commit.author.date);

            const hoursAgo = (now - lastCommitDate) / (1000 * 60 * 60);

            const message = lastCommit.commit.message.toLowerCase();


            // Analyze commit patterns and generate appropriate response

            if (recent.length >= 5) {

                return {

                    message: `Whoa! ${recent.length} commits in 24 hours? You're absolutely on fire! 🔥`,

                    mood: "excited"

                };

            } else if (recent.length >= 2) {

                return {

                    message: `Nice! ${recent.length} commits today. Your consistency is paying off! 💪`,

                    mood: "celebrating"

                };

            } else if (hoursAgo < 2) {

                // Recent commit - analyze the message

                if (message.includes('fix') || message.includes('bug')) {

                    return {

                        message: "Great debugging work! Every bug fixed makes you a better developer! 🔍",

                        mood: "encouraging"

                    };

                } else if (message.includes('feature') || message.includes('add')) {

                    return {

                        message: "Love seeing new features! Your project is evolving beautifully! ✨",

                        mood: "celebrating"

                    };

                } else if (message.includes('refactor') || message.includes('clean')) {

                    return {

                        message: "Refactoring shows real craftsmanship! Clean code is happy code! 🌟",

                        mood: "celebrating"

                    };

                } else {

                    return {

                        message: `Just saw your commit: "${lastCommit.commit.message.substring(0, 30)}..." - nice work! 🎯`,

                        mood: "encouraging"

                    };

                }

            } else if (hoursAgo < 24) {

                return {

                    message: `Your last commit was ${Math.round(hoursAgo)} hours ago. Ready for another coding session? 💻`,

                    mood: "nudging"

                };

            } else if (hoursAgo < 72) {

                return {

                    message: "It's been a couple days since your last commit. No pressure, but your code misses you! 🤔",

                    mood: "nudging"

                };

            } else {

                return {

                    message: "Hey there, stranger! Your repo has been quiet. Time to wake it up with some fresh code? ⚡",

                    mood: "worried"

                };

            }

        }


        // Enhanced GitHub check function

        async function checkGitHub() {

            const repo = document.getElementById('repoInput').value.trim();

            

            if (!repo || !repo.includes('/')) {

                showMessage("Please enter a valid GitHub repo in the format: username/repository", "worried");

                return;

            }


            updateStatus(`Checking ${repo}...`);

            

            try {

                const commits = await fetchGitHubCommits(repo);

                const analysis = analyzeCommits(commits);

                

                showMessage(analysis.message, analysis.mood);

                updateStatus(`Analyzed ${commits.length} commits from ${repo}`);

                

                // Store the repo for future checks

                localStorage.setItem('lastRepo', repo);

                

            } catch (error) {

                if (error.message.includes('404')) {

                    showMessage("Hmm, I can't find that repository. Double-check the username and repo name? 🤔", "worried");

                } else if (error.message.includes('403')) {

                    showMessage("Looks like I hit a rate limit. Try again in a few minutes! ⏰", "nudging");

                } else {

                    showMessage("Having trouble connecting to GitHub right now. Check your internet connection? 📡", "worried");

                }

                updateStatus("GitHub check failed");

                console.error('Detailed error:', error);

            }

        }


        // Auto-check functionality

        let autoCheckInterval = null;


        function startAutoCheck() {

            const repo = document.getElementById('repoInput').value.trim();

            if (!repo || !repo.includes('/')) {

                showMessage("Please enter a valid repo first, then I can auto-check it for you! 📝", "nudging");

                return;

            }


            if (autoCheckInterval) {

                clearInterval(autoCheckInterval);

                autoCheckInterval = null;

                updateStatus("Auto-check stopped");

                showMessage("I'll stop auto-checking for now. Click again to restart! 😊", "encouraging");

                return;

            }


            autoCheckInterval = setInterval(() => {

                checkGitHub();

            }, 5 * 60 * 1000); // 5 minutes


            showMessage("I'll check your repo every 5 minutes now! Keep coding! 🚀", "excited");

            updateStatus("Auto-checking every 5 minutes...");

        }


        function checkMyRepos() {
